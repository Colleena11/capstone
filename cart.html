function updateTotals(items) { const subtotal = Array.isArray(items) ? items.reduce((sum, item) => sum + item.price, 0) : 0; const shipping = subtotal > 0 ? 500 : 0; // Changed shipping to 500 pesos const total = subtotal + shipping; document.getElementById('subtotal').textContent = `₱{subtotal.toFixed(2)}`; document.getElementById('shipping').textContent = `₱{shipping.toFixed(2)}`; document.getElementById('total').textContent = `₱{total.toFixed(2)}`; }
